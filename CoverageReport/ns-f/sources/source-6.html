670                 if (e.isSiblingTo(edge)) {
671                     e.siblings.add(edge);
672                 }
673             }

676             //oriented the opposite way of the middle one (do that for all sibling lists)

680                 }
681             }
682

684             for (Edge edge : e.getSource().getIncidentEdges()) {
685                 if (edge == e) {
686                     continue;
687                 }

689                     edge.siblings = e.siblings;
690                 }
691             }
692

694
