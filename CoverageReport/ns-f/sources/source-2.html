303         double offset = 0;
304         int index = siblings.indexOf(this);
305         //offset both edges orthogonally, offsets differently when both face same direction
306         if (siblings.size() == 2) {
307             offset = 0.5 * multiEdgeOffset;
308             if (index == 1) {
309                 if (siblings.get(0).sameOrientationAs(this)) {
310                     offset *= -1;
311                 }
312             }
313 
314         }
315         if (siblings.size() == 3) {
316             if (index == 1) {
317                 offset = 0;
318             } else if (index == 0) {
319                 offset = multiEdgeOffset;
320             } else if (index == 2) {
321                 offset = (siblings.get(0).sameOrientationAs(this) ? -1 : 1) * multiEdgeOffset;
322             }
323         }
324         if (siblings.size() == 4) {
325             int sameOrientationCount = 0;
326             double offsetMultiplier;
327             for (int i = 0; i < siblings.size(); i++) {
328                 if (i == index) {
329                     int correctedOffsetCounter = (sameOrientationCount >= 2 ? -(i - 1) : (sameOrientationCount + 1));
330                     if (Math.abs(correctedOffsetCounter) > 1) {
331                         offsetMultiplier = 0.75;
332                     } else {
333                         offsetMultiplier = 0.5;
334                     }
335                     offset = offsetMultiplier * correctedOffsetCounter * multiEdgeOffset;
336                     break;
337                 }
338                 if (siblings.get(i).sameOrientationAs(this)) {
339                     sameOrientationCount++;
340                 }
341 
342             }
343         }
344         return offset;

404      * Checks if a given vector (x, y) is within a close margin of the
405      * sharp edge (the form of the edge is given by its control points)
406      *
407      * @param x x coordinate of the position to check
408      * @param y y coordinate of the position to check
409      * @return -1 if edge does not contain position, i>=0 otherwise (where [i-1, i] are the indices
410      * of the control points that have been hit). If i==0, then [i-1] is the source vertex
411      */
412     private int containsCoordinateSharp(double x, double y) {
413         if (controlPoints.size() == 0) {
414             return containsCoordinateFlat(x, y);
415         }
416         double dist = Vector2D.distancePointToLine(x, y, source.getCoordinates(), controlPoints.get(0).getPosition());
417 
418         if (dist < multiEdgeOffset * 0.5) {
419             return 0;
420         }
421         for (int i = 1; i < controlPoints.size(); i++) {
422             Vector2D a = controlPoints.get(i - 1).getPosition();
423             Vector2D b = controlPoints.get(i).getPosition();
424             if (Vector2D.distancePointToLine(x, y, a, b) < multiEdgeOffset * 0.5) {
425                 return i;
426             }
427         }
428         Vector2D last = controlPoints.get(controlPoints.size() - 1).getPosition();
429         dist = Vector2D.distancePointToLine(x, y, last, target.getCoordinates());
430 
431         return dist < multiEdgeOffset * 0.5 ? controlPoints.size() : -1;
432     }
