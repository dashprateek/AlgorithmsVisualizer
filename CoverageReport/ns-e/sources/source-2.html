66         //first, do a classification by angle between center and lineStart
67         a = a.minus(center);
68         b = b.minus(center);
69         double alpha = a.theta();
70         double w = sizeBox.width / 2;
71         double h = sizeBox.height / 2;
72 
73         Vector2D r;
74         Vector2D dir;
75 
76         double coeff;
77 
78         Vector2D diff = a.minus(b);
79         final double counterRotation = Math.toRadians(diff.theta());
80         /*
81          * The shape consists of 4 lines that can be identified by the angle of a point that lies
82          * on the edge of the shape. An edge point with the angle 45 is on the first line,
83          * an edge point with angle 110 lies on the second etc..
84          *
85          *
86          * Consider a point A outside of the diamond with an angle of <90 (w.r.t. center)
87          *
88          * Now, if the internal point B is at the center, the intersection point will always be
89          * at the first line. If however the point B is below the center and A has a very flat angle
90          * then the intersection could occur at the 4th line.
91          *
92          * This correction is being calculated with newC1/C2 and depending on the result, we classify
93          * the correct line.
94          *
95          * */
96 
97         int lineNumber;
98 
99         if (alpha < 90) {
100             double newC1 = (new Vector2D(w, 0)).rotate(b, -counterRotation).getY();
101             double newC2 = (new Vector2D(0, h)).rotate(b, -counterRotation).getY();
102 
103             if (newC1 > b.getY()) {
104                 lineNumber = 3;
105             } else if (newC2 < b.getY()) {
106                 lineNumber = 1;
107             } else {
108                 lineNumber = 0;
109             }
110         } else if (alpha < 180) {
111             double newC1 = (new Vector2D(0, h)).rotate(b, -counterRotation).getY();
112             double newC2 = (new Vector2D(-w, 0)).rotate(b, -counterRotation).getY();
113 
114             if (newC1 > b.getY()) {
115                 lineNumber = 0;
116             } else if (newC2 < b.getY()) {
117                 lineNumber = 2;
118             } else {
119                 lineNumber = 1;
120             }
121 
122         } else if (alpha < 270) {
123             double newC1 = (new Vector2D(-w, 0)).rotate(b, -counterRotation).getY();
124             double newC2 = (new Vector2D(0, -h)).rotate(b, -counterRotation).getY();
125 
126             if (newC1 > b.getY()) {
127                 lineNumber = 1;
128             } else if (newC2 < b.getY()) {
129                 lineNumber = 3;
130             } else {
131                 lineNumber = 2;
132             }
133 
134         } else {
135             double newC1 = (new Vector2D(0, -h)).rotate(b, -counterRotation).getY();
136             double newC2 = (new Vector2D(w, 0)).rotate(b, -counterRotation).getY();
137 
138             if (newC1 > b.getY()) {
139                 lineNumber = 2;
140             } else if (newC2 < b.getY()) {
141                 lineNumber = 0;
142             } else {
143                 lineNumber = 3;
144             }
145 
146         }
147         LineParameters l = getParamsFromNumber(lineNumber, w, h);
148         r = l.r;
149         dir = l.dir;
150 
151         coeff = dir.getY() / dir.getX();
152         double t = ((a.getY() - r.getY()) - coeff * (a.getX() - r.getX()))
153                 / (-b.getY() + a.getY() - coeff * (-b.getX() + a.getX()));
154 
155         return a.plus(b.minus(a).multiply(t)).plus(center); //line defined as a + (b-a)t
156     }
